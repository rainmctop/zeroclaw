3/// A single help entry: one or more keys that trigger the same action.
4///
5/// The renderer joins keys with " / " so you don't have to format manually.
6/// An entry with all-empty keys/action renders as a blank spacer row.
7#[derive(Debug, Clone, Default)]
8pub struct HelpEntry {
9    /// Keys that trigger this action, e.g. ["↑", "k"].
10    pub keys: Vec<String>,
11    /// Human-readable description of the action.
12    pub action: String,
13}
14
15impl HelpEntry {
16    pub fn new<K: Into<String>>(
17        keys: impl IntoIterator<Item = K>,
18        action: impl Into<String>,
19    ) -> Self {
20        Self {
21            keys: keys.into_iter().map(Into::into).collect(),
22            action: action.into(),
23        }
24    }
25
26    /// Convenience: single key.
27    pub fn key(key: impl Into<String>, action: impl Into<String>) -> Self {
28        Self {
29            keys: vec![key.into()],
30            action: action.into(),
31        }
32    }
33
34    /// Blank spacer row.
35    pub fn spacer() -> Self {
36        Self {
37            keys: vec![],
38            action: String::new(),
39        }
40    }
41
42    /// Keyless description row (no key column, just text).
43    pub fn desc(action: impl Into<String>) -> Self {
44        Self {
45            keys: vec![],
46            action: action.into(),
47        }
48    }
49
50    /// Format keys as "↑ / k" etc.
51    pub fn key_str(&self) -> String {
52        self.keys.join(" / ")
53    }
54}

56/// A node in the help context tree.
57///
58/// The help system cascades: Pane → Tab → Widget (or Screen → Tab → Widget
59/// for the config pane). Each level produces one `HelpNode`. The modal renders
60/// them depth-first:
61///
62///   [title]
63///   [description, soft-wrapped]
64///   key   action
65///   key   action
66///   ── dim separator ──
67///   [child title]
68///   ...
69///
70/// Any field may be empty/None — the renderer skips it cleanly.
71#[derive(Debug, Clone, Default)]
72pub struct HelpNode {
73    /// Short label shown as a dim section header (e.g. "Tab", "Widget"). None = no header.
74    pub title: Option<String>,
75    /// Prose description shown above the keybindings, soft-wrapped to modal width.
76    pub description: Option<String>,
77    /// Keybinding entries for this level.
78    pub entries: Vec<HelpEntry>,
79    /// Child nodes (tab-level, widget-level, etc.).
80    pub children: Vec<HelpNode>,
81}
82
83impl HelpNode {
84    /// Leaf node with just keybindings.
85    pub fn entries(entries: Vec<HelpEntry>) -> Self {
86        Self {
87            entries,
88            ..Default::default()
89        }
90    }
91
92    /// Consume self and append a child node, returning the modified node.
93    pub fn with_child(mut self, child: HelpNode) -> Self {
94        self.children.push(child);
95        self
96    }
97}

99/// Implement this on any struct that can contribute to the help modal.
100pub trait HelpContext {
101    fn help_context(&self) -> HelpNode;
102}

112/// A one-row context-window usage bar.
113///
114/// Renders left-aligned into whatever `Rect` you hand it.
115/// Returns `None` from `widget()` when there is nothing to show.
116pub struct CtxBar {
117    pub input_tokens: Option<u64>,
118    pub max_tokens: Option<u64>,
119}
120
121impl CtxBar {
122    pub fn new(input_tokens: Option<u64>, max_tokens: Option<u64>) -> Self {
123        Self {
124            input_tokens,
125            max_tokens,
126        }
127    }
128
129    /// `true` when there is something worth rendering.
130    pub fn has_content(&self) -> bool {
131        self.input_tokens.is_some() || self.max_tokens.is_some()
132    }
133
134    /// Build a `Paragraph` widget, or `None` if there is nothing to show.
135    pub fn widget(&self) -> Option<Paragraph<'static>> {
136        let (text, pct_opt) = match (self.input_tokens, self.max_tokens) {
137            (Some(used), Some(max)) if max > 0 => {
138                let pct = (used as f64 / max as f64 * 100.0).min(100.0);
139                let bar_width: usize = 16;
140                let filled = ((pct / 100.0) * bar_width as f64).round() as usize;
141                let empty = bar_width.saturating_sub(filled);
142                let bar = format!(
143                    "[{}{}]",
144                    "\u{2588}".repeat(filled),
145                    "\u{2591}".repeat(empty)
146                );
147                let label = format!(
148                    " ctx: {:>7} / {:>7}  {}  {:.0}%",
149                    fmt_tokens(used),
150                    fmt_tokens(max),
151                    bar,
152                    pct,
153                );
154                (label, Some(pct))
155            }
156            (Some(used), None) => {
157                let label = format!(" ctx: {} tokens", fmt_tokens(used));
158                (label, None)
159            }
160            _ => return None,
161        };
162
163        let color = match pct_opt {
164            Some(p) if p >= 90.0 => Color::Red,
165            Some(p) if p >= 75.0 => Color::Yellow,
166            _ => Color::DarkGray,
167        };
168
169        Some(Paragraph::new(Line::from(Span::styled(
170            text,
171            Style::default().fg(color),
172        ))))
173    }
174}

176fn fmt_tokens(n: u64) -> String {
177    let s = n.to_string();
178    let mut out = String::with_capacity(s.len() + s.len() / 3);
179    for (i, ch) in s.chars().rev().enumerate() {
180        if i > 0 && i % 3 == 0 {
181            out.push(',');
182        }
183        out.push(ch);
184    }
185    out.chars().rev().collect()
186}
